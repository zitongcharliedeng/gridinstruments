# Mount Top Bar

Mounts the SolidJS `TopBar` component into `#topbar-mount`. This is the bridge
between the vanilla TS app and the Solid component. The `mountTopBar` function
is called from app-core during initialization.

The `onReset` callback clears all `gi_*` localStorage keys and reloads the page,
matching the existing imperative handler in `main.ts`. The `onAbout` callback
opens the about dialog via the `dialogMachine` actor passed in.

``` {.typescript file=components/mount-topbar.tsx}
import { render } from 'solid-js/web';
import { TopBar } from './TopBar';

export function mountTopBar(
  mountEl: HTMLElement,
  onAbout: () => void,
  onReset: () => void,
): void {
  render(
    () => (
      <TopBar
        onAbout={onAbout}
        onReset={onReset}
      />
    ),
    mountEl,
  );
}
```
